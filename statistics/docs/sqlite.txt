# Parse blk*.dat into blk*.db
Get-ChildItem -Path ../bitcoin-data/blocks/blk*.dat | ForEach-Object {
    Get-Date
    Write-Host $_

    $dbName = "$($_.BaseName).db"

    if (-not Test-Path $dbName) {
        $command = @"
        CREATE TABLE IF NOT EXISTS block (
            hash BLOB PRIMARY KEY ON CONFLICT IGNORE,
            time INTEGER
        );

        CREATE TABLE IF NOT EXISTS tx (
            block_hash BLOB REFERENCES block(hash),
            hash BLOB PRIMARY KEY ON CONFLICT IGNORE
        );

        CREATE TABLE IF NOT EXISTS out (
            tx_hash BLOB REFERENCES tx(hash),
            count INTEGER,
            amount INTEGER,
            pub_key BLOB,
            UNIQUE(tx_hash, count) ON CONFLICT IGNORE
        );
"@

        Write-Output "$command" | sqlite3 $dbName
    }
}

Get-ChildItem -Path ../bitcoin-data/blocks/blk*.dat | ForEach-Object {

    $dbName = "$($_.BaseName).db"

    $command = "BEGIN TRANSACTION;`n";
    $command += "PRAGMA foreign_keys = ON;`n"
    $command += ./crawl $_;
    $command += "COMMIT;";
    Write-Output "$command" | sqlite3 $dbName
}

# Transaction sizes
FROM out
JOIN tx ON out.tx_hash = tx.hash
JOIN block ON tx.block_hash = block.hash
SELECT out.amount, block.time
ORDER BY block.time ASC;

OR

SELECT 
    out.amount,
    block.time
FROM 
    out
    JOIN tx ON out.tx_hash = tx.hash
    JOIN block ON tx.block_hash = block.hash
ORDER BY 
    block.time ASC;

OR

SELECT
  count(out_table.amount),
  sum(out_table.amount)
FROM
  out AS out_table
JOIN
  tx AS tx_table
  ON out_table.tx_hash = tx_table.hash
JOIN
  block AS block_table
  ON tx_table.block_hash = block_table.hash
WHERE block_table.time>1303603200 and block_table.time<1303682632 and out_table.amount>100000000000;


# This does the trick
SELECT
    DATE(block.time, 'unixepoch') AS day,
    CASE
        WHEN out.amount <       100000      THEN '0-0.001'
        WHEN out.amount <      1000000      THEN '0.001-0.01'
        WHEN out.amount <     10000000      THEN '0.01-0.1'
        WHEN out.amount <    100000000      THEN '0.1-1'
        WHEN out.amount <   1000000000      THEN '1-10'
        WHEN out.amount <  10000000000      THEN '10-100'
        WHEN out.amount < 100000000000      THEN '100-1000'
        ELSE '>=1000'
    END AS size_bin,
    COUNT(*) AS count,
    SUM(out.amount)/100000000
FROM out
JOIN tx ON out.tx_hash = tx.hash
JOIN block ON tx.block_hash = block.hash
GROUP BY day, size_bin
ORDER BY day ASC, size_bin;
